5        if len(arr1d.shape) >= 2:  # 이차원 배열 이상인 경우

7            exit()  # 코드 종료
8        
9        if isinstance(arr1d[0], np.int32) or isinstance(arr1d[0], np.int64) or isinstance(arr1d[0], np.float64):
10           pass  # arr1d[0]이 int나 float인 경우는 넘기고
11       else:  # 그 외인 경우는
12           print("정렬 불가")  # 정렬 불가 출력후

21
